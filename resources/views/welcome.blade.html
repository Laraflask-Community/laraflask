⚡ Quick Start
# Clone the starter app
git clone https://github.com/laraflask/laraflask-app my-app
cd my-app && pip install -r requirements.txt

# Setup environment
cp .env.example .env
python artisan.py key:generate

# Run migrations and seed
python artisan.py migrate
python artisan.py db:seed

# Start development server
python artisan.py serve
